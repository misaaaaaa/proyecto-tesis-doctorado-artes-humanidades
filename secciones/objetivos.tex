\chapter{Objetivos de la investigación}

% máximo media página

\section{Objetivos generales}

\begin{enumerate}
    \item Construir un glosario de prácticas de síntesis popular de sintetizadores sonoros, usando herramientas textuales, metafóricas, electrónicas y computacionales, para promover nuevas formas de luthería popular y artesanal.
    \item Diseñar y fabricar infraestructura electrónica para la sintesíntesis, incluyendo la creación de estándares, estrategias y módulos fundamentales que permitan crear nuevos instrumentos.
    \item  Diseñar y fabricar infraestructura computacional para la sintesíntesis, incluyendo la creación de bibliotecas de código que permitan crear nuevos instrumentos.
\end{enumerate}

\section{Objetivos específicos}

\begin{enumerate}
    \item Proponer nuevas estrategias electrónicas de síntesis sonora, incluyendo el uso de circuitos integrados, de construcción de placas de circuito impreso, y de documentación de procesos de diseño electrónico.
    \item Proponer nuevas estrategias computacionales de síntesis sonora, incluyendo el uso de lenguajes de programación, diseño y programación de nuevas bibliotecas de código, y documentación de procesos de diseño computacional.
    \item Proponer nuevas estrategias populares de síntesis sonora, incluyendo su documentación, trabajo público, y construcción a partir de materiales de fácil adquisición, reparables y sostenibles.
\end{enumerate}
