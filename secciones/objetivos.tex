\chapter{Objetivos de la investigación}

% máximo media página

\section{Objetivos generales}

\begin{enumerate}
    \item Construir un glosario de prácticas de síntesis popular de sintetizadores sonoros, usando herramientas textuales, metafóricas, electrónicas y computacionales, para promover nuevas formas de expresión sonora a través de la luthería de nuevos instrumentos.
    \item Diseñar y fabricar infraestructura electrónica y computacional para la síntesis de sintetizadores sonoros bajo los principios de popularidad y artesanía.
\end{enumerate}

\section{Objetivos específicos}

\begin{enumerate}
    \item Proponer nuevas estrategias de síntesis sonora, incluyendo su documentación, construcción y parametrización.
    \item Construir módulos fundamentales de la síntesis sonora, usando electrónica y computación.
\end{enumerate}
